def enhance(image):
    temp = np.zeros((image.shape), dtype='uint8')
    minn = image.min()
    maxx = image.max()
    for i in range(image.shape[0]):
        for j in range(image.shape[1]):
            temp[i, j] = int(255 * (image[i,j] - minn) / (maxx - minn))
    return temp
